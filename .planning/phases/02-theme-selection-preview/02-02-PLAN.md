**Test function structure:**
```
function runCarouselTests() {
  // Use the same test markdown as runParserTests
  var testMarkdown = [same string as in runParserTests];
  var parsed = parseMarkdown(testMarkdown);

  // Temporarily render results to test DOM
  renderResults(parsed, 'test-carousel.md');

  var errors = [];

  // Test 1: 8 theme swatches exist
  var swatches = document.querySelectorAll('.theme-swatch');
  if (swatches.length !== 8) errors.push('Expected 8 swatches, got ' + swatches.length);

  // Test 2: Editorial swatch is pre-selected
  var selected = document.querySelector('.theme-swatch.selected');
  if (!selected || selected.getAttribute('data-theme-id') !== 'editorial')
    errors.push('Editorial swatch not pre-selected');

  // Test 3: slide-canvas has data-theme="editorial"
  var canvas = document.querySelector('.slide-canvas');
  if (!canvas || canvas.getAttribute('data-theme') !== 'editorial')
    errors.push('slide-canvas missing data-theme="editorial"');

  // Test 4: Correct number of slide-frame elements
  var frames = document.querySelectorAll('.slide-frame');
  if (frames.length !== parsed.slides.length)
    errors.push('Expected ' + parsed.slides.length + ' slide frames, got ' + frames.length);

  // Test 5: First slide is active
  var activeSlide = document.querySelector('.slide-frame.active');
  if (!activeSlide || activeSlide.getAttribute('data-slide-index') !== '0')
    errors.push('First slide not active');

  // Test 6: Slide types correct (cover, body, body, body, body, cta for 6 slides)
  if (frames.length === 6) {
    if (!frames[0].querySelector('.slide-type-cover')) errors.push('Slide 0 not cover type');
    if (!frames[1].querySelector('.slide-type-body')) errors.push('Slide 1 not body type');
    if (!frames[5].querySelector('.slide-type-cta')) errors.push('Slide 5 not cta type');
  }

  // Test 7: Navigation arrows exist
  var arrows = document.querySelectorAll('.carousel-arrow');
  if (arrows.length !== 2) errors.push('Expected 2 navigation arrows, got ' + arrows.length);

  // Test 8: Dot indicators match slide count
  var dots = document.querySelectorAll('.carousel-dot');
  if (dots.length !== parsed.slides.length)
    errors.push('Expected ' + parsed.slides.length + ' dots, got ' + dots.length);

  // Test 9: Slide counter shows "1 / 6"
  var counter = document.querySelector('.slide-counter');
  if (!counter || counter.textContent.trim() !== '1 / ' + parsed.slides.length)
    errors.push('Counter text wrong: got "' + (counter ? counter.textContent : 'null') + '"');

  // Test 10: Theme switch works - change to deep-space and verify
  if (canvas) {
    canvas.setAttribute('data-theme', 'deep-space');
    var computedBg = getComputedStyle(frames[0]).backgroundColor;
    // Reset back to editorial
    canvas.setAttribute('data-theme', 'editorial');
    // We can't easily compare computed colors across browsers, so just verify attribute changed
    if (canvas.getAttribute('data-theme') !== 'editorial')
      errors.push('Theme switch back to editorial failed');
  }

  // Test 11: goToSlide function exists
  if (typeof goToSlide !== 'function' && !document.querySelector('.carousel-arrow'))
    errors.push('goToSlide function or navigation not found');

  // Test 12: Caption and hashtags still render
  if (!document.querySelector('.caption-block')) errors.push('Caption block missing');
  if (!document.querySelector('.hashtags-container')) errors.push('Hashtags container missing');

  // Clean up: reset to upload view
  var uploadNew = document.querySelector('.upload-new-link');
  if (uploadNew) uploadNew.click();

  if (errors.length === 0) {
    console.log('CAROUSEL TESTS: ALL PASSED (' + 12 + ' checks)');
  } else {
    console.error('CAROUSEL TESTS FAILED:');
    errors.forEach(function(e) { console.error('  - ' + e); });
  }
  return errors;
}
```
